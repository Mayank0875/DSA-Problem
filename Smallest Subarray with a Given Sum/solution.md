int minSubArrayLen(int target, vector<int>& nums) {
    int n = nums.size();
    if (n == 0) {
        return 0;
    }
    int ans = INT_MAX;
    int left = 0;
    int sum = 0;
    for (int i = 0; i < n; i++) {
        sum += nums[i];
        while (sum >= target) {
            ans = min(ans, i + 1 - left);
            sum -= nums[left++];
        }
    }
    return (ans != INT_MAX) ? ans : 0;
}

    for (int i = 0; i < n; i++) {

            if (len < ans)
                ans = len;
            sum -= nums[left++];

    let ans = Number.MAX_SAFE_INTEGER;

def minSubArrayLen(target, nums):
    n = len(nums)
    if n == 0:
        return 0

    ans = float('inf')
    left = 0
    total = 0

    for i in range(n):
        total += nums[i]
        while total >= target:
            ans = min(ans, i + 1 - left)
            total -= nums[left]
            left += 1

    return ans if ans != float('inf') else 0
